  <script>
    function placeBet() {
      const winRate = 0.3; // نسبة الفوز (30%)
      const random = Math.random();
      const resultDiv = document.getElementById("result");

      if (random < winRate) {
        resultDiv.textContent = "🎉 مبروك! ربحت 10 عملات!";
        resultDiv.style.color = "green";
      } else {
        resultDiv.textContent = "😢 للأسف، لم تربح هذه المرة.";
        resultDiv.style.color = "red";
      }
    }
  </script>
